  \uncover<1->{입력 프로그램 $P$가 주어졌을 때, $P$의 실행 의미는 보존하면서 요약해석 결과는 더 커지게 하는 변환 $\tau(P)$를 만들자}

  \vspace{20pt}
  
  \uncover<2->{\includegraphics[scale=0.3]{img/목표.png}}

\end{frame}

\begin{frame}
  \frametitle{연구 주제와 비교}
  \uncover<1->{non-relational domain 가정. 실행 의미는 $\top$이 아니지만, 요약해석 결과는 $\top$이 되는 프로그램 $P$를 만들자(특정 변수 x에 대해서)}
  \begin{align*}
    \uncover<2->{(\alpha(\llbracket P \rrbracket \mathbb{S}))\texttt{x} &\neq \top} \\
    \uncover<2->{(\llbracket P \rrbracket^{\mathcal{A}}\alpha(\mathbb{S}))\texttt{x} &= \top} \\
    \uncover<3->{\alpha(\llbracket P \rrbracket \mathbb{S}) &< \llbracket P \rrbracket^{\mathcal{A}}\alpha(\mathbb{S})}
  \end{align*}

  \uncover<4->{\includegraphics[scale=0.3]{img/목표.png}}
\end{frame}

\section{한계점 및 발전 방향}
\begin{frame}
  \frametitle{한계점}

  계속 ``존재성''만 얘기하고, 증명하고 있음
  구체적으로 어떻게 프로그램을 변환하거나 찾는지에 대한 내용이 전혀 없음

  \begin{itemize}
    \uncover<2->{\item $S$를 어떻게 찾을 것인가?}
    \uncover<3->{\item In$(S)$를 어떻게 프로그램으로 구현할 것인가?}
  \end{itemize}
  
  \uncover<4->{0부터 합성하는 것이 아니라, 기존에 있던 프로그램을 변환시키는 방법을 사용할 것이라면 이 방법을 찾는 것이 목표}

  \uncover<5->{혹은, 기계적으로는 \textcolor{red}{불가능}하다는 것을 증명하는 것이 목표}

\end{frame}

\begin{frame}
  \frametitle{발전 방향}
  \begin{itemize}
    \item \uncover<1->{골탕 합성 연구의 기반은 Rice theorem에서 나온 존재성}
    \item \uncover<2->{그러나 여전히 합성이 기계적으로 가능하다는 보장을 가지고 출발한 것은 아님}
    \item \uncover<3->{이 논문의 아이디어를 응용, if문으로 구성된 기본 틀을 주고, 그 안을 합성하게 하는 방법을 생각해 보자}
    \item \uncover<4->{+, Rice theorem을 직접적으로 활용하기 위해 while문을 기본 틀로 주는 것도 가능}
  \end{itemize}
